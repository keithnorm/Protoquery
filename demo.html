Basic Prototype element methods

$('#demo1').insert({bottom: 'this is some inserted text'});

new Element()

var p = new Element('p', {
  'class': 'some_class'
}
}).update('hi everybody!');

$('#demo2').insert(p);

new Ajax.Request

new Ajax.Request('response.html', {
  onSuccess: function(transport) {
    $('#demo3').insert(transport.responseText);
  }
});

Array extensions

[100, 300, 400, 200].max();

[{name: 'jack'}, {name: 'jill'}].map(
  function(obj, i) { 
    return obj.name;
  });

=> ['jack', 'jill']
Or... 
[{name: 'jack'}, {name: 'jill'}].pluck('name');

=> ['jack', 'jill']
